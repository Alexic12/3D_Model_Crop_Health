#!/usr/bin/env bash
set -euo pipefail

trap 'kill -TERM $(jobs -p) 2>/dev/null || true' TERM INT

# API on 8000
uvicorn app.api.main:app --host 0.0.0.0 --port 8000 &

# Streamlit UI 1 on 8501
streamlit run app/ui/main.py --server.port=8501 --server.address=0.0.0.0 &

# Streamlit UI 2 on 8502
streamlit run app/ui2/main.py --server.port=8502 --server.address=0.0.0.0 &

# If any dies, exit so ECS replaces the task
wait -n
exit $?

# Step 1: Create new task definition JSON file
cat > streamlit-task.json <<EOF
{
  "family": "streamlit-task",
  "networkMode": "awsvpc",
  "requiresCompatibilities": ["FARGATE"],
  "cpu": "1024",
  "memory": "2048",
  "executionRoleArn": "arn:aws:iam::147997127433:role/ecsTaskExecutionRole",
  "containerDefinitions": [
    {
      "name": "streamlit-app",
      "image": "147997127433.dkr.ecr.us-east-1.amazonaws.com/agrorisk:latest",
      "essential": true,
      "portMappings": [
        { "containerPort": 8000, "protocol": "tcp" },
        { "containerPort": 8501, "protocol": "tcp" },
        { "containerPort": 8502, "protocol": "tcp" }
      ],
      "logConfiguration": {
        "logDriver": "awslogs",
        "options": {
          "awslogs-group": "/ecs/streamlit-task",
          "awslogs-region": "us-east-1",
          "awslogs-stream-prefix": "ecs"
        }
      }
    }
  ]
}
EOF

# Step 2: Register new task definition
aws ecs register-task-definition --cli-input-json file://streamlit-task.json

# Step 3: Get the latest revision number (optional but recommended)
aws ecs describe-task-definition --task-definition streamlit-task --query "taskDefinition.revision"

# Step 4: Update ECS service to use the latest revision and force new deployment
aws ecs update-service \
  --cluster streamlit-cluster \
  --service streamlit-task-service-uu9sz6hc \
  --task-definition streamlit-task \
  --force-new-deployment
